#### Day 3：时间轴组件

**重点任务**：
```javascript
// js/components/timeline.js

class Timeline {
    constructor(container, data) {
        this.container = container;
        this.data = data;
    }
    
    render() {
        // 渲染时间轴
        const html = this.data.map(item => `
            <div class="timeline-item">
                <div class="timeline-dot"></div>
                <div class="timeline-content">
                    <h3>${item.location}</h3>
                    <p>${item.action}</p>
                    <time>${this.formatDate(item.timestamp)}</time>
                </div>
            </div>
        `).join('');
        
        this.container.innerHTML = html;
    }
    
    formatDate(timestamp) {
        return new Date(timestamp).toLocaleString('zh-CN');
    }
}
```

#### Day 4：对话流组件

**重点任务**：
```javascript
// js/components/chat.js

class ChatFlow {
    constructor(container, emails) {
        this.container = container;
        this.emails = emails;
        this.displayCount = 5;
    }
    
    render() {
        // 渲染对话
        const displayEmails = this.emails.slice(0, this.displayCount);
        
        const html = displayEmails.map(email => `
            <div class="chat-message ${email.direction}">
                <div class="avatar">${email.direction === 'sent' ? 'P' : 'X'}</div>
                <div class="message-content">
                    <p>${email.content}</p>
                    <time>${this.formatDate(email.timestamp)}</time>
                </div>
            </div>
        `).join('');
        
        this.container.innerHTML = html;
    }
    
    loadMore() {
        this.displayCount += 5;
        this.render();
    }
}
```

| 决策点 | 建议方案 | 备选方案 |

### 六、风险与应对

| 风险 | 影响 | 应对措施 |
|------|------|----------|
| **API 无法调用** | 无法获取实时数据 | 使用模拟数据 |
| **跨域问题** | API 调用失败 | 使用 CORS 代理或放弃实时调用 |
| **性能问题** | 页面加载慢 | 图片压缩、代码优化 |
| **浏览器兼容** | 部分功能失效 | 渐进增强、Polyfill |
